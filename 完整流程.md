# 资金流入流出预测 — 完整实战流程

## 〇、方案总览

```
数据入库(MySQL) → 数据清洗 → 天级聚合 → 特征工程(SQL) → 训练特征表
                                                                 ↓
提交文件 ← 加权融合 ← 递归预测(Python) ← 模型训练(LGB+XGB+Ridge)
```

**核心设计决策：**

| 决策项 | 选择 | 理由 |
|--------|------|------|
| 建模粒度 | 预测每日总量 | 子成分做辅助特征，不单独建模 |
| 预测策略 | 递归预测（逐日推进） | 短期lag特征对精度至关重要 |
| 模型方案 | LightGBM + XGBoost + Ridge 加权融合 | 多样性融合降低方差 |
| 特征工程 | SQL为主完成 | 充分利用窗口函数，Python端代码简洁 |
| 验证策略 | 2014-08月（31天）做验证集 | 模拟真实预测场景 |

**评价指标要点：**
- 每日相对误差：`Purchase_i = |真实值 - 预测值| / 真实值`
- 误差=0 → 10分，误差≥0.3 → 0分，单调递减
- **总分 = 申购得分×45% + 赎回得分×55%**（赎回更重要）
- 优化重点：避免任何一天误差>30%（直接0分），赎回精度优先

## 二、天级聚合（SQL）

### 2.1 核心日聚合表

将用户级明细聚合为每日全市场总量，同时提取子成分和活跃用户数。

```sql
DROP TABLE IF EXISTS daily_agg;
CREATE TABLE daily_agg AS
SELECT
    c.report_date,
    -- ===== 目标变量 =====
    COALESCE(SUM(b.total_purchase_amt), 0)  AS y_purchase,
    COALESCE(SUM(b.total_redeem_amt), 0)    AS y_redeem,

    -- ===== 申购子成分 =====
    COALESCE(SUM(b.direct_purchase_amt), 0) AS sub_direct_purchase,
    COALESCE(SUM(b.purchase_bal_amt), 0)    AS sub_purchase_bal,
    COALESCE(SUM(b.purchase_bank_amt), 0)   AS sub_purchase_bank,

    -- ===== 赎回子成分 =====
    COALESCE(SUM(b.consume_amt), 0)         AS sub_consume,
    COALESCE(SUM(b.transfer_amt), 0)        AS sub_transfer,
    COALESCE(SUM(b.tftobal_amt), 0)         AS sub_tftobal,
    COALESCE(SUM(b.tftocard_amt), 0)        AS sub_tftocard,

    -- ===== 其他汇总 =====
    COALESCE(SUM(b.share_amt), 0)           AS total_share,
    COALESCE(SUM(b.category1), 0)           AS total_cat1,
    COALESCE(SUM(b.category2), 0)           AS total_cat2,
    COALESCE(SUM(b.category3), 0)           AS total_cat3,
    COALESCE(SUM(b.category4), 0)           AS total_cat4,

    -- ===== 活跃用户统计 =====
    COUNT(DISTINCT b.user_id)               AS active_users,
    COUNT(DISTINCT CASE WHEN b.total_purchase_amt > 0
                        THEN b.user_id END) AS purchase_users,
    COUNT(DISTINCT CASE WHEN b.total_redeem_amt > 0
                        THEN b.user_id END) AS redeem_users,

    -- ===== 余额统计 =====
    COALESCE(SUM(b.tBalance), 0)            AS total_balance

FROM calendar c
LEFT JOIN user_balance b ON c.report_date = b.report_date
WHERE c.report_date <= '2014-08-31'
GROUP BY c.report_date
ORDER BY c.report_date;

ALTER TABLE daily_agg ADD PRIMARY KEY (report_date);
```

## 三、特征工程（SQL 核心）

### 3.1 日历特征

```sql
DROP TABLE IF EXISTS feat_calendar;
CREATE TABLE feat_calendar AS
SELECT
    report_date,
    DAYOFWEEK(report_date)  AS dayofweek,       -- 1=周日 ... 7=周六
    CASE WHEN DAYOFWEEK(report_date) IN (1, 7)
         THEN 1 ELSE 0 END AS is_weekend,
    DAY(report_date)        AS day_of_month,
    MONTH(report_date)      AS month,
    QUARTER(report_date)    AS quarter,
    CEIL(DAY(report_date) / 7.0) AS week_of_month,

    -- 月初（1~3号）
    CASE WHEN DAY(report_date) <= 3
         THEN 1 ELSE 0 END AS is_month_start,
    -- 月末（倒数3天内）
    CASE WHEN DAY(report_date) >= DAY(LAST_DAY(report_date)) - 2
         THEN 1 ELSE 0 END AS is_month_end,
    -- 季末
    CASE WHEN MONTH(report_date) IN (3,6,9,12)
          AND DAY(report_date) >= DAY(LAST_DAY(report_date)) - 2
         THEN 1 ELSE 0 END AS is_quarter_end,
    -- 发薪日（常见10号、15号、25号前后）
    CASE WHEN DAY(report_date) IN (10,15,25)
         THEN 1 ELSE 0 END AS is_salary_day

FROM calendar;

ALTER TABLE feat_calendar ADD PRIMARY KEY (report_date);
```

### 3.2 节假日特征

```sql
DROP TABLE IF EXISTS feat_holiday;
CREATE TABLE feat_holiday AS
SELECT
    c.report_date,
    CASE WHEN h.holiday_date IS NOT NULL THEN 1 ELSE 0 END AS is_holiday,

    -- 距下一个假期的天数（0=当天就是假期）
    (SELECT MIN(DATEDIFF(h2.holiday_date, c.report_date))
     FROM holidays h2
     WHERE h2.holiday_date >= c.report_date)  AS days_to_next_holiday,

    -- 距上一个假期的天数
    (SELECT MIN(DATEDIFF(c.report_date, h3.holiday_date))
     FROM holidays h3
     WHERE h3.holiday_date <= c.report_date)  AS days_since_last_holiday

FROM calendar c
LEFT JOIN holidays h ON c.report_date = h.holiday_date;

ALTER TABLE feat_holiday ADD PRIMARY KEY (report_date);
```

### 3.3 滞后特征

使用窗口函数 LAG，基于 daily_agg 的连续日期行。

```sql
DROP TABLE IF EXISTS feat_lag;
CREATE TABLE feat_lag AS
SELECT
    report_date,

    -- ===== 申购滞后 =====
    LAG(y_purchase, 1)  OVER w AS p_lag1,
    LAG(y_purchase, 2)  OVER w AS p_lag2,
    LAG(y_purchase, 3)  OVER w AS p_lag3,
    LAG(y_purchase, 7)  OVER w AS p_lag7,
    LAG(y_purchase, 14) OVER w AS p_lag14,
    LAG(y_purchase, 21) OVER w AS p_lag21,
    LAG(y_purchase, 28) OVER w AS p_lag28,
    LAG(y_purchase, 30) OVER w AS p_lag30,

    -- ===== 赎回滞后 =====
    LAG(y_redeem, 1)  OVER w AS r_lag1,
    LAG(y_redeem, 2)  OVER w AS r_lag2,
    LAG(y_redeem, 3)  OVER w AS r_lag3,
    LAG(y_redeem, 7)  OVER w AS r_lag7,
    LAG(y_redeem, 14) OVER w AS r_lag14,
    LAG(y_redeem, 21) OVER w AS r_lag21,
    LAG(y_redeem, 28) OVER w AS r_lag28,
    LAG(y_redeem, 30) OVER w AS r_lag30

FROM daily_agg
WINDOW w AS (ORDER BY report_date);

ALTER TABLE feat_lag ADD PRIMARY KEY (report_date);
```

### 3.4 滑窗统计特征

```sql
DROP TABLE IF EXISTS feat_rolling;
CREATE TABLE feat_rolling AS
SELECT
    report_date,

    -- ===== 申购滑窗均值 =====
    AVG(y_purchase) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS p_roll7_mean,
    AVG(y_purchase) OVER (ORDER BY report_date
        ROWS BETWEEN 14 PRECEDING AND 1 PRECEDING) AS p_roll14_mean,
    AVG(y_purchase) OVER (ORDER BY report_date
        ROWS BETWEEN 30 PRECEDING AND 1 PRECEDING) AS p_roll30_mean,

    -- ===== 申购滑窗标准差 =====
    STDDEV(y_purchase) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS p_roll7_std,
    STDDEV(y_purchase) OVER (ORDER BY report_date
        ROWS BETWEEN 30 PRECEDING AND 1 PRECEDING) AS p_roll30_std,

    -- ===== 申购滑窗最大最小 =====
    MAX(y_purchase) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS p_roll7_max,
    MIN(y_purchase) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS p_roll7_min,

    -- ===== 赎回滑窗均值 =====
    AVG(y_redeem) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS r_roll7_mean,
    AVG(y_redeem) OVER (ORDER BY report_date
        ROWS BETWEEN 14 PRECEDING AND 1 PRECEDING) AS r_roll14_mean,
    AVG(y_redeem) OVER (ORDER BY report_date
        ROWS BETWEEN 30 PRECEDING AND 1 PRECEDING) AS r_roll30_mean,

    -- ===== 赎回滑窗标准差 =====
    STDDEV(y_redeem) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS r_roll7_std,
    STDDEV(y_redeem) OVER (ORDER BY report_date
        ROWS BETWEEN 30 PRECEDING AND 1 PRECEDING) AS r_roll30_std,

    -- ===== 赎回滑窗最大最小 =====
    MAX(y_redeem) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS r_roll7_max,
    MIN(y_redeem) OVER (ORDER BY report_date
        ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING)  AS r_roll7_min

FROM daily_agg;

ALTER TABLE feat_rolling ADD PRIMARY KEY (report_date);
```

### 3.5 同周几历史特征

捕捉"上周同一天""上上周同一天"的模式，对周期性行为非常重要。

```sql
DROP TABLE IF EXISTS feat_same_weekday;
CREATE TABLE feat_same_weekday AS
SELECT
    a.report_date,
    -- 同周几-1周前
    b1.y_purchase AS p_same_wd_1w,
    b1.y_redeem   AS r_same_wd_1w,
    -- 同周几-2周前
    b2.y_purchase AS p_same_wd_2w,
    b2.y_redeem   AS r_same_wd_2w,
    -- 同周几-3周前
    b3.y_purchase AS p_same_wd_3w,
    b3.y_redeem   AS r_same_wd_3w,
    -- 同周几-4周前
    b4.y_purchase AS p_same_wd_4w,
    b4.y_redeem   AS r_same_wd_4w,

    -- 同周几近4周均值
    (COALESCE(b1.y_purchase,0) + COALESCE(b2.y_purchase,0)
   + COALESCE(b3.y_purchase,0) + COALESCE(b4.y_purchase,0))
   / NULLIF(
       (b1.y_purchase IS NOT NULL) + (b2.y_purchase IS NOT NULL)
     + (b3.y_purchase IS NOT NULL) + (b4.y_purchase IS NOT NULL), 0)
    AS p_same_wd_4w_mean,

    (COALESCE(b1.y_redeem,0) + COALESCE(b2.y_redeem,0)
   + COALESCE(b3.y_redeem,0) + COALESCE(b4.y_redeem,0))
   / NULLIF(
       (b1.y_redeem IS NOT NULL) + (b2.y_redeem IS NOT NULL)
     + (b3.y_redeem IS NOT NULL) + (b4.y_redeem IS NOT NULL), 0)
    AS r_same_wd_4w_mean

FROM daily_agg a
LEFT JOIN daily_agg b1 ON b1.report_date = DATE_SUB(a.report_date, INTERVAL  7 DAY)
LEFT JOIN daily_agg b2 ON b2.report_date = DATE_SUB(a.report_date, INTERVAL 14 DAY)
LEFT JOIN daily_agg b3 ON b3.report_date = DATE_SUB(a.report_date, INTERVAL 21 DAY)
LEFT JOIN daily_agg b4 ON b4.report_date = DATE_SUB(a.report_date, INTERVAL 28 DAY);

ALTER TABLE feat_same_weekday ADD PRIMARY KEY (report_date);
```

### 3.6 外部市场特征（收益率 + Shibor）

```sql
DROP TABLE IF EXISTS feat_market;
CREATE TABLE feat_market AS
SELECT
    c.report_date,

    -- 收益率（用最近已知值填充）
    COALESCE(i.mfd_daily_yield,
        (SELECT i2.mfd_daily_yield FROM mfd_interest i2
         WHERE i2.mfd_date <= c.report_date
         ORDER BY i2.mfd_date DESC LIMIT 1))      AS daily_yield,

    COALESCE(i.mfd_7daily_yield,
        (SELECT i3.mfd_7daily_yield FROM mfd_interest i3
         WHERE i3.mfd_date <= c.report_date
         ORDER BY i3.mfd_date DESC LIMIT 1))      AS weekly_yield,

    -- Shibor 隔夜和1周利率（关键指标）
    COALESCE(s.Interest_O_N,
        (SELECT s2.Interest_O_N FROM mfd_shibor s2
         WHERE s2.mfd_date <= c.report_date
         ORDER BY s2.mfd_date DESC LIMIT 1))      AS shibor_ON,

    COALESCE(s.Interest_1_W,
        (SELECT s3.Interest_1_W FROM mfd_shibor s3
         WHERE s3.mfd_date <= c.report_date
         ORDER BY s3.mfd_date DESC LIMIT 1))      AS shibor_1W,

    COALESCE(s.Interest_1_M,
        (SELECT s4.Interest_1_M FROM mfd_shibor s4
         WHERE s4.mfd_date <= c.report_date
         ORDER BY s4.mfd_date DESC LIMIT 1))      AS shibor_1M,

    COALESCE(s.Interest_3_M,
        (SELECT s5.Interest_3_M FROM mfd_shibor s5
         WHERE s5.mfd_date <= c.report_date
         ORDER BY s5.mfd_date DESC LIMIT 1))      AS shibor_3M

FROM calendar c
LEFT JOIN mfd_interest i ON c.report_date = i.mfd_date
LEFT JOIN mfd_shibor   s ON c.report_date = s.mfd_date;
```

### 3.7 用户行为结构特征

子成分占比和活跃用户占比等结构性信息。

```sql
DROP TABLE IF EXISTS feat_structure;
CREATE TABLE feat_structure AS
SELECT
    report_date,

    -- 申购子成分占比
    CASE WHEN y_purchase > 0
         THEN sub_direct_purchase / y_purchase ELSE 0 END AS ratio_direct_purchase,
    CASE WHEN y_purchase > 0
         THEN sub_purchase_bank / y_purchase ELSE 0 END   AS ratio_bank_purchase,

    -- 赎回子成分占比
    CASE WHEN y_redeem > 0
         THEN sub_consume / y_redeem ELSE 0 END           AS ratio_consume,
    CASE WHEN y_redeem > 0
         THEN sub_tftocard / y_redeem ELSE 0 END          AS ratio_tftocard,
    CASE WHEN y_redeem > 0
         THEN sub_tftobal / y_redeem ELSE 0 END           AS ratio_tftobal,

    -- 人均申购/赎回
    CASE WHEN purchase_users > 0
         THEN y_purchase / purchase_users ELSE 0 END      AS per_user_purchase,
    CASE WHEN redeem_users > 0
         THEN y_redeem / redeem_users ELSE 0 END          AS per_user_redeem,

    -- 申购/赎回用户占比
    CASE WHEN active_users > 0
         THEN purchase_users / active_users ELSE 0 END    AS ratio_purchase_users,
    CASE WHEN active_users > 0
         THEN redeem_users / active_users ELSE 0 END      AS ratio_redeem_users,

    -- category 汇总
    total_cat1, total_cat2, total_cat3, total_cat4

FROM daily_agg;

ALTER TABLE feat_structure ADD PRIMARY KEY (report_date);
```

### 3.8 趋势与变化率特征

```sql
DROP TABLE IF EXISTS feat_trend;
CREATE TABLE feat_trend AS
SELECT
    report_date,

    -- 申购：日环比变化率
    CASE WHEN LAG(y_purchase,1) OVER w > 0
         THEN (y_purchase - LAG(y_purchase,1) OVER w)
              / LAG(y_purchase,1) OVER w
         ELSE 0 END AS p_daily_change_rate,

    -- 申购：周环比变化率
    CASE WHEN LAG(y_purchase,7) OVER w > 0
         THEN (y_purchase - LAG(y_purchase,7) OVER w)
              / LAG(y_purchase,7) OVER w
         ELSE 0 END AS p_weekly_change_rate,

    -- 赎回：日环比变化率
    CASE WHEN LAG(y_redeem,1) OVER w > 0
         THEN (y_redeem - LAG(y_redeem,1) OVER w)
              / LAG(y_redeem,1) OVER w
         ELSE 0 END AS r_daily_change_rate,

    -- 赎回：周环比变化率
    CASE WHEN LAG(y_redeem,7) OVER w > 0
         THEN (y_redeem - LAG(y_redeem,7) OVER w)
              / LAG(y_redeem,7) OVER w
         ELSE 0 END AS r_weekly_change_rate,

    -- 申购/赎回比值
    CASE WHEN y_redeem > 0
         THEN y_purchase / y_redeem ELSE 0 END AS purchase_redeem_ratio

FROM daily_agg
WINDOW w AS (ORDER BY report_date);

ALTER TABLE feat_trend ADD PRIMARY KEY (report_date);
```

```python
def relative_error(y_true, y_pred):
    """每日相对误差"""
    return np.abs(y_true - y_pred) / np.where(y_true == 0, 1, y_true)

def competition_score(y_true, y_pred):
    """
    模拟比赛评分：
    - 每天误差=0 → 10分，误差>=0.3 → 0分
    - 使用线性近似（真实函数未公开）
    """
    re = relative_error(y_true, y_pred)
    daily_scores = np.maximum(0, 10 * (1 - re / 0.3))
    return daily_scores.sum(), daily_scores

def full_score(y_true_p, y_pred_p, y_true_r, y_pred_r):
    """总分 = 申购*45% + 赎回*55%"""
    score_p, _ = competition_score(y_true_p, y_pred_p)
    score_r, _ = competition_score(y_true_r, y_pred_r)
    total = score_p * 0.45 + score_r * 0.55
    return total, score_p, score_r

def train_lgb(X_train, y_train, X_valid, y_valid, label='purchase'):
    """LightGBM 训练"""
    params = {
        'objective': 'regression',
        'metric': 'mape',
        'boosting_type': 'gbdt',
        'learning_rate': 0.05,
        'num_leaves': 31,
        'max_depth': 6,
        'min_child_samples': 10,
        'subsample': 0.8,
        'colsample_bytree': 0.8,
        'reg_alpha': 0.1,
        'reg_lambda': 0.1,
        'verbose': -1,
        'n_jobs': -1,
        'seed': 42
    }
    dtrain = lgb.Dataset(X_train, y_train)
    dvalid = lgb.Dataset(X_valid, y_valid, reference=dtrain)

    model = lgb.train(
        params, dtrain,
        num_boost_round=2000,
        valid_sets=[dvalid],
        callbacks=[
            lgb.early_stopping(100),
            lgb.log_evaluation(200)
        ]
    )
    print(f"[LGB-{label}] best iteration: {model.best_iteration}")
    return model

def train_xgb(X_train, y_train, X_valid, y_valid, label='purchase'):
    """XGBoost 训练"""
    params = {
        'objective': 'reg:squarederror',
        'eval_metric': 'mape',
        'learning_rate': 0.05,
        'max_depth': 6,
        'min_child_weight': 10,
        'subsample': 0.8,
        'colsample_bytree': 0.8,
        'reg_alpha': 0.1,
        'reg_lambda': 1.0,
        'seed': 42,
        'verbosity': 0
    }
    dtrain = xgb.DMatrix(X_train, y_train)
    dvalid = xgb.DMatrix(X_valid, y_valid)

    model = xgb.train(
        params, dtrain,
        num_boost_round=2000,
        evals=[(dvalid, 'valid')],
        early_stopping_rounds=100,
        verbose_eval=200
    )
    print(f"[XGB-{label}] best iteration: {model.best_iteration}")
    return model

def train_ridge(X_train, y_train, label='purchase'):
    """Ridge 回归（需标准化）"""
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X_train.fillna(0))
    model = Ridge(alpha=100.0)
    model.fit(X_scaled, y_train)
    print(f"[Ridge-{label}] R²: {model.score(X_scaled, y_train):.4f}")
    return model, scaler

# ----- 训练申购模型 -----

# =====================================================
# 八、9月递归预测
# =====================================================

def build_calendar_features(dt):
    """构造单日日历+节假日特征"""
    holidays_sep = {6, 7, 8}  # 2014年9月中秋
    dow = dt.isoweekday()  # 1=Mon ... 7=Sun
    mysql_dow = dow % 7 + 1   # 转为 MySQL DAYOFWEEK: 1=Sun,2=Mon...7=Sat
    dom = dt.day
    last_day = pd.Timestamp(dt.year, dt.month, 1) + pd.offsets.MonthEnd(0)
    last_dom = last_day.day
    return {
        'dayofweek': mysql_dow,
        'is_weekend': 1 if mysql_dow in (1, 7) else 0,
        'day_of_month': dom,
        'month': dt.month,
        'quarter': (dt.month - 1) // 3 + 1,
        'week_of_month': int(np.ceil(dom / 7)),
        'is_month_start': 1 if dom <= 3 else 0,
        'is_month_end': 1 if dom >= last_dom - 2 else 0,
        'is_quarter_end': 1 if dt.month in (3,6,9,12) and dom >= last_dom-2
                          else 0,
        'is_salary_day': 1 if dom in (10, 15, 25) else 0,
        'is_holiday': 1 if dom in holidays_sep else 0,
        'days_to_next_holiday': min(d - dom for d in holidays_sep
                                    if d >= dom) if any(d >= dom
                                    for d in holidays_sep) else 30,
        'days_since_last_holiday': min(dom - d for d in holidays_sep
                                       if d <= dom) if any(d <= dom
                                       for d in holidays_sep) else 30,
    }

# 获取8月历史数据用于lag计算
hist = df_clean[['report_date','y_purchase','y_redeem']].copy()
hist = hist.set_index('report_date').sort_index()

# 最后一行的市场特征（9月carry forward）
last_market = df_clean.iloc[-1][[
    'daily_yield','weekly_yield','shibor_ON','shibor_1W','shibor_1M','shibor_3M'
]].to_dict()

# 最后一行的结构特征（carry forward）
last_structure = df_clean.iloc[-1][[
    'ratio_direct_purchase','ratio_bank_purchase',
    'ratio_consume','ratio_tftocard','ratio_tftobal',
    'per_user_purchase','per_user_redeem',
    'ratio_purchase_users','ratio_redeem_users',
    'total_cat1','total_cat2','total_cat3','total_cat4'
]].to_dict()

pred_dates = pd.date_range('2014-09-01', '2014-09-30')
results = []

for dt in pred_dates:
    row = {}

    # 1) 日历 + 节假日
    row.update(build_calendar_features(dt))

    # 2) 滞后特征（从历史 + 已预测值中取）
    for lag_days, suffix in [(1,'1'),(2,'2'),(3,'3'),(7,'7'),
                              (14,'14'),(21,'21'),(28,'28'),(30,'30')]:
        lag_dt = dt - pd.Timedelta(days=lag_days)
        if lag_dt in hist.index:
            row[f'p_lag{suffix}'] = hist.loc[lag_dt, 'y_purchase']
            row[f'r_lag{suffix}'] = hist.loc[lag_dt, 'y_redeem']
        else:
            row[f'p_lag{suffix}'] = 0
            row[f'r_lag{suffix}'] = 0

    # 3) 滑窗统计（基于历史序列，包含已预测值）
    for window, name in [(7,'7'),(14,'14'),(30,'30')]:
        window_dates = pd.date_range(dt - pd.Timedelta(days=window),
                                     dt - pd.Timedelta(days=1))
        p_vals = [hist.loc[d,'y_purchase'] for d in window_dates
                  if d in hist.index]
        r_vals = [hist.loc[d,'y_redeem'] for d in window_dates
                  if d in hist.index]
        row[f'p_roll{name}_mean'] = np.mean(p_vals) if p_vals else 0
        row[f'r_roll{name}_mean'] = np.mean(r_vals) if r_vals else 0
        if name == '7':
            row[f'p_roll7_std'] = np.std(p_vals) if len(p_vals)>1 else 0
            row[f'r_roll7_std'] = np.std(r_vals) if len(r_vals)>1 else 0
            row[f'p_roll7_max'] = max(p_vals) if p_vals else 0
            row[f'p_roll7_min'] = min(p_vals) if p_vals else 0
            row[f'r_roll7_max'] = max(r_vals) if r_vals else 0
            row[f'r_roll7_min'] = min(r_vals) if r_vals else 0
        if name == '30':
            row[f'p_roll30_std'] = np.std(p_vals) if len(p_vals)>1 else 0
            row[f'r_roll30_std'] = np.std(r_vals) if len(r_vals)>1 else 0

    # 4) 同周几特征
    for wk, suffix in [(1,'1w'),(2,'2w'),(3,'3w'),(4,'4w')]:
        wd_dt = dt - pd.Timedelta(weeks=wk)
        if wd_dt in hist.index:
            row[f'p_same_wd_{suffix}'] = hist.loc[wd_dt, 'y_purchase']
            row[f'r_same_wd_{suffix}'] = hist.loc[wd_dt, 'y_redeem']
        else:
            row[f'p_same_wd_{suffix}'] = 0
            row[f'r_same_wd_{suffix}'] = 0
    p_wd_vals = [row[f'p_same_wd_{s}'] for s in ['1w','2w','3w','4w']
                 if row[f'p_same_wd_{s}'] > 0]
    r_wd_vals = [row[f'r_same_wd_{s}'] for s in ['1w','2w','3w','4w']
                 if row[f'r_same_wd_{s}'] > 0]
    row['p_same_wd_4w_mean'] = np.mean(p_wd_vals) if p_wd_vals else 0
    row['r_same_wd_4w_mean'] = np.mean(r_wd_vals) if r_wd_vals else 0

    # 5) 市场特征（carry forward）
    row.update(last_market)

    # 6) 结构特征（carry forward）
    row.update(last_structure)

    # 7) 趋势特征
    yesterday = dt - pd.Timedelta(days=1)
    last_week = dt - pd.Timedelta(days=7)
    yp = hist.loc[yesterday,'y_purchase'] if yesterday in hist.index else 0
    yr = hist.loc[yesterday,'y_redeem']   if yesterday in hist.index else 0
    wp = hist.loc[last_week,'y_purchase'] if last_week in hist.index else 0
    wr = hist.loc[last_week,'y_redeem']   if last_week in hist.index else 0
    row['p_daily_change_rate'] = (yp - row.get('p_lag2',0)) / max(row.get('p_lag2',1), 1)
    row['p_weekly_change_rate'] = (yp - wp) / max(wp, 1)
    row['r_daily_change_rate'] = (yr - row.get('r_lag2',0)) / max(row.get('r_lag2',1), 1)
    row['r_weekly_change_rate'] = (yr - wr) / max(wr, 1)
    row['purchase_redeem_ratio'] = yp / max(yr, 1)

    # ----- 预测 -----
    row_p = pd.DataFrame([{k: row[k] for k in PURCHASE_FEATURE_COLS}])
    row_r = pd.DataFrame([{k: row[k] for k in REDEEM_FEATURE_COLS}])

    p_lgb = lgb_final_p.predict(row_p)[0]
    p_xgb = xgb_final_p.predict(xgb.DMatrix(row_p))[0]
    p_ridge = ridge_final_p.predict(
        scaler_final_p.transform(row_p.fillna(0)))[0]

    r_lgb = lgb_final_r.predict(row_r)[0]
    r_xgb = xgb_final_r.predict(xgb.DMatrix(row_r))[0]
    r_ridge = ridge_final_r.predict(
        scaler_final_r.transform(row_r.fillna(0)))[0]

    # 加权融合
    w1, w2, w3 = best_wp
    pred_purchase = max(0, w1*p_lgb + w2*p_xgb + w3*p_ridge)
    pred_redeem   = max(0, w1*r_lgb + w2*r_xgb + w3*r_ridge)

    # 更新历史序列（递归）
    hist.loc[dt] = [pred_purchase, pred_redeem]

    results.append({
        'report_date': dt.strftime('%Y%m%d'),
        'purchase': int(round(pred_purchase)),
        'redeem': int(round(pred_redeem))
    })

    print(f"{dt.strftime('%Y-%m-%d')} | "
          f"purchase={int(pred_purchase):>12,} | "
          f"redeem={int(pred_redeem):>12,}")

# =====================================================
# 九、输出提交文件
# =====================================================

submit = pd.DataFrame(results)
submit.to_csv('comp_predict_table.csv', index=False, header=False)
print(f"\n提交文件已生成: comp_predict_table.csv")
print(submit.head())
```

### Optuna 超参搜索示例

```python
import optuna

def objective(trial):
    params = {
        'objective': 'regression',
        'metric': 'mape',
        'learning_rate': trial.suggest_float('lr', 0.01, 0.1),
        'num_leaves': trial.suggest_int('num_leaves', 15, 63),
        'max_depth': trial.suggest_int('max_depth', 4, 10),
        'min_child_samples': trial.suggest_int('min_child', 5, 50),
        'subsample': trial.suggest_float('subsample', 0.6, 1.0),
        'colsample_bytree': trial.suggest_float('colsample', 0.6, 1.0),
        'reg_alpha': trial.suggest_float('alpha', 1e-3, 10, log=True),
        'reg_lambda': trial.suggest_float('lambda', 1e-3, 10, log=True),
        'verbose': -1
    }
    dtrain = lgb.Dataset(X_train_r, y_train_r)
    dvalid = lgb.Dataset(X_valid_r, y_valid_r, reference=dtrain)
    model = lgb.train(params, dtrain, 1000,
                      valid_sets=[dvalid],
                      callbacks=[lgb.early_stopping(50),
                                 lgb.log_evaluation(0)])
    pred = model.predict(X_valid_r)
    score, _ = competition_score(y_valid_r.values, pred)
    return score

study = optuna.create_study(direction='maximize')
```
